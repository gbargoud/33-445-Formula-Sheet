	\section*{Operator Properties}
	\begin{align*}
		\expect{\op{A}}&=\bra{\psi}\op{A}\ket{\psi} &
		\Delta\op{A}&=\sqrt{\expect{\op{A}^2}-\expect{\op{A}}^2} &
		\op{U}\opt{U}&=1 & 
		[\op{A},\op{B}]&=\op{A}\op{B}-\op{B}\op{A} &
		\Delta\op{A}\Delta\op{B}&\ge\expect{\frac{[\op{A},\op{B}]}{2i}}
	\end{align*}
